USE Homework3DB;
CREATE TABLE Customer (
    Customer_ID INT AUTO_INCREMENT NOT NULL,
    Name VARCHAR(40) NOT NULL,
    Address VARCHAR(40),
    PRIMARY KEY (Customer_ID)
);

CREATE TABLE CreditCard (
    CC_No VARCHAR(16) NOT NULL,
    FullName VARCHAR(40) NOT NULL,
    ExpiryDate DATE NOT NULL,
    Verif_Code INT,
    CardType VARCHAR(18) NOT NULL,
    Cust_ID INT,
    PRIMARY KEY (CC_No),
    FOREIGN KEY (Cust_ID) REFERENCES Customer(Customer_ID)
);

CREATE TABLE Orders (
    Order_ID INT AUTO_INCREMENT NOT NULL,
    Customer_ID INT NOT NULL,
    CC_No VARCHAR(16) NOT NULL,
    OrderDate DATE NOT NULL,
    OrderTime TIME NOT NULL,
    Amount DECIMAL(10,2),
    PRIMARY KEY (Order_ID),
    FOREIGN KEY (Customer_ID) REFERENCES Customer(Customer_ID),
    FOREIGN KEY (CC_No) REFERENCES CreditCard(CC_No)
);

CREATE TABLE DigitalProduct (
    Product_ID INT AUTO_INCREMENT NOT NULL,
    Title VARCHAR(50),
    Price DECIMAL(10,2),
    Description VARCHAR(225),
    ProductType VARCHAR(15),
    PRIMARY KEY (Product_ID)
);

CREATE TABLE Order_line (
    Order_ID INT NOT NULL,
    Product_ID INT NOT NULL,
    Quantity INT,
    PRIMARY KEY (Order_ID, Product_ID),
    FOREIGN KEY (Order_ID) REFERENCES Orders(Order_ID),
    FOREIGN KEY (Product_ID) REFERENCES DigitalProduct(Product_ID)
);

CREATE TABLE Movie (
    Movie_ID INT NOT NULL,
    Director VARCHAR(50),
    Rating VARCHAR(10),
    PRIMARY KEY (Movie_ID),
    FOREIGN KEY (Movie_ID) REFERENCES DigitalProduct(Product_ID)
);

CREATE TABLE Music (
    Music_ID INT NOT NULL,
    Genre VARCHAR(30),
    Album VARCHAR(30),
    PRIMARY KEY (Music_ID),
    FOREIGN KEY (Music_ID) REFERENCES DigitalProduct(Product_ID)
);

CREATE TABLE EBook (
    Ebook_ID INT NOT NULL,
    First_Author VARCHAR(60),
    PageCount INT,
    PRIMARY KEY (Ebook_ID),
    FOREIGN KEY (Ebook_ID) REFERENCES DigitalProduct(Product_ID)
);
